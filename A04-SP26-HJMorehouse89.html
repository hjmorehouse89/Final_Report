import pandas as pd

spark = SparkSession.builder.appName("LightcastModule5").getOrCreate()
df = spark.read.option("header", "true") \
    .option("inferSchema", "true") \
    .option("multiLine", "true") \
    .option("escape", "\"") \
    .csv("./data/lightcast_job_postings.csv")

print("Loaded data")
#df.printSchema()

continuous_features = [
    "MIN_YEARS_EXPERIENCE",
    "MAX_YEARS_EXPERIENCE",
    "DURATION",
    "SALARY_FROM",
    "SALARY_TO",
]

categorical_features = [
    "EDUCATION_LEVELS_NAME",
    "REMOTE_TYPE_NAME"
]


target = "SALARY"
selected_cols = continuous_features + categorical_features + [target]
df_clean = df.select(*selected_cols)



df_clean.groupBy("MIN_YEARS_EXPERIENCE")\
           .agg(F.count("*").alias("count")) \
           .orderBy(F.col("MIN_YEARS_EXPERIENCE").desc())\
           .show(truncate=False)

df_clean = df_clean.withColumn("REMOTE_TYPE",
when(trim(col("REMOTE_TYPE_NAME"))== "Remote", "Remote")
.when(trim(col("REMOTE_TYPE_NAME"))== "Hybrid Remote", "Hybrid")
.when(trim(col("REMOTE_TYPE_NAME"))== "Not Remote", "Onsite")
.when(col("REMOTE_TYPE_NAME").isNull(),"Onsite")
.otherwise("Onsite"))

df_clean.select("REMOTE_TYPE_NAME").distinct().show(truncate=False)
df_clean.drop("REMOTE_TYPE_NAME")


df_clean = df_clean.withColumn("EXPERIENCE_GROUP",
when(col("MIN_YEARS_EXPERIENCE").between(0,1),"Internship/Entry Level")
.when(col("MIN_YEARS_EXPERIENCE").isNull(),"Junior")
.when(col("MIN_YEARS_EXPERIENCE").between(1,3),"Junior")
.when(col("MIN_YEARS_EXPERIENCE").between(3,5),"Mid-Level")
.when(col("MIN_YEARS_EXPERIENCE").between(5,10),"Senior")
.otherwise("Expert")
)

df_clean.show(5,truncate=False)

df_clean = df_clean.withColumn(
    "SALARY_CLEAN",
    F.when((F.col("SALARY").isNull()) | (F.col("SALARY") == 0) | F.isnan("SALARY"), None).otherwise(F.col("SALARY"))
).withColumn(
    "SALARY_FROM_CLEAN",
    F.when((F.col("SALARY_FROM").isNull()) | (F.col("SALARY_FROM") == 0) | F.isnan("SALARY_FROM"), None).otherwise(F.col("SALARY_FROM"))
).withColumn(
    "SALARY_TO_CLEAN",
    F.when((F.col("SALARY_TO").isNull()) | (F.col("SALARY_TO") == 0) | F.isnan("SALARY_TO"), None).otherwise(F.col("SALARY_TO"))
)
df_clean = df_clean.withColumn(
    "SALARY_FINAL",
    F.when(F.col("SALARY_CLEAN").isNotNull(),F.col("SALARY_CLEAN"))
    .when(
        F.col("SALARY_FROM_CLEAN").isNotNull() & F.col("SALARY_TO_CLEAN").isNotNull(),(F.col("SALARY_FROM_CLEAN") + F.col("SALARY_TO_CLEAN"))/2
    )
    .when(
        F.col("SALARY_FROM_CLEAN").isNotNull() & F.col("SALARY_TO_CLEAN").isNull(), F.col("SALARY_FROM_CLEAN")
    )
    .when(
        F.col("SALARY_TO_CLEAN").isNotNull() & F.col("SALARY_FROM_CLEAN").isNull(), F.col("SALARY_TO_CLEAN")
    )
    .otherwise(None)
)


for c in selected_cols:
    df_clean = df_clean.filter(col(c).isNotNull())

print("Cleaned row count:", df_clean.count())
df_clean = df_clean.withColumn(
    "MIN_YEARS_EXPERIENCE_SQ",
    pow(col("MIN_YEARS_EXPERIENCE"), 2)
)

indexers = [
    StringIndexer(inputCol=c, outputCol=f"{c}_idx", handleInvalid="keep")
    for c in categorical_features
]

encoders = [
    OneHotEncoder(inputCol=f"{c}_idx", outputCol=f"{c}_vec")
    for c in categorical_features
]

features_input = continuous_features + [f"{c}_vec" for c in categorical_features]

assembler = VectorAssembler(
    inputCols=features_input,
    outputCol="features"
)

poly_input = [
    "MIN_YEARS_EXPERIENCE",
    "MIN_YEARS_EXPERIENCE_SQ",
    "MAX_YEARS_EXPERIENCE",
    "DURATION",
    "SALARY_FROM"
] + [f"{c}_vec" for c in categorical_features]

poly_assembler = VectorAssembler(

    outputCol="features_poly"
)

pipeline = Pipeline(stages=indexers + encoders + [assembler, poly_assembler])
pipeline_model = pipeline.fit(df_clean)
final_df = pipeline_model.transform(df_clean)

final_df.select("SALARY", "features", "features_poly").show(5, truncate=False)


train_df, test_df = final_df.randomSplit([0.8, 0.2], seed=42)

print("Train count:", train_df.count())
print("Test count:", test_df.count())


lr = LinearRegression(
    featuresCol="features",
    labelCol="SALARY"
)

lr_model = lr.fit(train_df)
lr_predictions = lr_model.transform(test_df)


poly_lr = LinearRegression(
    featuresCol="features_poly",
    labelCol="SALARY"
)

rf = RandomForestRegressor(
    featuresCol="features",
    labelCol="SALARY",
    numTrees=200,
    maxDepth=8,
    seed=42
)

rf_model = rf.fit(train_df)

rmse_eval = RegressionEvaluator(labelCol="SALARY", predictionCol="prediction", metricName="rmse")
r2_eval = RegressionEvaluator(labelCol="SALARY", predictionCol="prediction", metricName="r2")
mae_eval = RegressionEvaluator(labelCol="SALARY", predictionCol="prediction", metricName="mae")

def evaluate_model(name, predictions, model_summary=None, num_params=None):
    rmse = rmse_eval.evaluate(predictions)
    r2 = r2_eval.evaluate(predictions)
    mae = mae_eval.evaluate(predictions)

    aic = None
    bic = None

    if model_summary is not None:
        try:
            aic = model_summary.aic
        except:
            aic = None

        try:
            n = model_summary.numInstances
            deviance = model_summary.deviance
            dispersion = model_summary.dispersion
            k = num_params if num_params is not None else len(model_summary.coefficients) + 1

            log_likelihood = -0.5 * (
                n * math.log(2 * math.pi)
                + n * math.log(dispersion)
                + deviance / dispersion
            )
            bic = k * math.log(n) - 2 * log_likelihood
        except:
            bic = None

    return {
        "Model": name,
        "RMSE": rmse,
        "R2": r2,
        "MAE": mae,
        "AIC": aic,
        "BIC": bic
    }

lr_results = evaluate_model(
    "Linear Regression",
    lr_predictions,
    lr_model.summary,
    len(lr_model.coefficients) + 1
)

poly_results = evaluate_model(
    "Polynomial Regression",
    poly_predictions,
    poly_model.summary,
    len(poly_model.coefficients) + 1
)

rf_results = evaluate_model(
    "Random Forest",
    rf_predictions,
    None,
    None
)

results_df = pd.DataFrame([lr_results, poly_results, rf_results])
print(results_df)
